# Docker Compose Unificado - Projeto FIDC
# Serviços compartilhados entre todos os microserviços

version: '3.8'

services:
  # Redis compartilhado para sessões e cache
  redis:
    image: redis:7.2-alpine
    container_name: fidc-redis
    restart: unless-stopped
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    command: redis-server --appendonly yes --maxmemory 256mb --maxmemory-policy allkeys-lru
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 3
    networks:
      - fidc-network

  # Prometheus para métricas centralizadas
  prometheus:
    image: prom/prometheus:v2.48.0
    container_name: fidc-prometheus
    restart: unless-stopped
    ports:
      - "9090:9090"
    volumes:
      - prometheus_data:/prometheus
    command:
      - '--config.file=/etc/prometheus/prometheus.yml'
      - '--storage.tsdb.path=/prometheus'
      - '--web.enable-lifecycle'
    networks:
      - fidc-network

  # Grafana para visualização de métricas
  grafana:
    image: grafana/grafana:10.2.0
    container_name: fidc-grafana
    restart: unless-stopped
    ports:
      - "3000:3000"
    environment:
      - GF_SECURITY_ADMIN_PASSWORD=admin123
    volumes:
      - grafana_data:/var/lib/grafana
    depends_on:
      - prometheus
    networks:
      - fidc-network

volumes:
  redis_data:
    driver: local
  prometheus_data:
    driver: local
  grafana_data:
    driver: local

networks:
  fidc-network:
    driver: bridge
    name: fidc-network
